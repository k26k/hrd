멤버, 보드 테이블 생성
create table t_member(
	memberid	varchar(20),
    passwd		varchar(20) not null,
    name		varchar(30) not null,
    gender		varchar(10),
    joindate	timestamp default now(), -- 생성시 현재 날짜
    primary key(memberid) -- 기본키 
);

create table t_board(
    bnum		int primary key auto_increment, -- 자동 증가
    title		varchar(100) not null,
    content		text not null,
    regdate		timestamp default now(),
    memberid	varchar(20),
    constraint fk_member_board foreign key (memberid) references t_member(memberid) on delete cascade -- 참조키 설정
);
